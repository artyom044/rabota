#include <iostream>
#include <string>
#include <algorithm> // Для std::min и std::max

int main() {
    double x, y;
    int dayNumber;

    // Ввод данных для вычисления
    std::cout << "Введите x: ";
    std::cin >> x;
    std::cout << "Введите y: ";
    std::cin >> y;

    // Вычисление выражения
    double result = (std::min(0.0, x) - std::min(0.0, y)) / (std::max(y, x) * std::max(y, x));
    std::cout << "Результат вычисления: " << result << std::endl;

    // Ввод номера дня недели
    std::cout << "Введите номер дня недели (1-7): ";
    std::cin >> dayNumber;

    // Название дня недели по номеру
    std::string dayName;
    switch(dayNumber) {
        case 1: dayName = "Monday"; break;
        case 2: dayName = "Tuesday"; break;
        case 3: dayName = "Wednesday"; break;
        case 4: dayName = "Thursday"; break;
        case 5: dayName = "Friday"; break;
        case 6: dayName = "Saturday"; break;
        case 7: dayName = "Sunday"; break;
        default: dayName = "Invalid day number"; break;
    }

    std::cout << "День недели: " << dayName << std::endl;

    return 0;
}
