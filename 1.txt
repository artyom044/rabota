#include <iostream>   // подключает библиотеку для работы с вводом (cin) и выводом (cout)
#include <cmath>      // подключает математическую библиотеку, чтобы использовать функции sin() и константы M_PI.
using namespace std;   // избавляет от необходимости писать std:: перед cin, cout, sin и т.п.

int main() { // точка входа программы.
    double a, A, B, C;   // объявляет переменные для стороны и трёх углов.
    cin >> a >> A >> B >> C;  // ввод с клавиатуры: сторона и три угла в градусах

    double rad = M_PI / 180;   // создаём переменную для перевода градусов в радианы. (π / 180) радиана rad

    A *= rad;   // преобразуем угол A из градусов в радианы.
    B *= rad;   // преобразуем угол B.
    C *= rad;   // преобразуем угол C.

    // Вычисление площади по формуле:
    // S = 0.5 * a^2 * sin(B) * sin(C) / sin(A)
    cout << 0.5 * a * a * sin(B) * sin(C) / sin(A); // выводит результат — площадь треугольника по формуле.
    return 0;
}
